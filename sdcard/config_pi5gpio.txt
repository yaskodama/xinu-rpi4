# config.txt — Xinu Pi 5 first-light, GPIO14/15 serial + HDMI.
#
# To restore Raspberry Pi OS: copy config.txt.linux.bak over config.txt.

arm_64bit=1
kernel=kernel_xinu.img
kernel_address=0x80000
auto_initramfs=0

# Serial on the 40-pin header GPIO14/15 (pins 8/10).  On the Pi 5 these
# are RP1's UART0, reached through the PCIe outbound window.  Two things
# are required for a bare-metal kernel to use it:
#   pciex4_reset=0   -> leave the PCIe root complex configured by the
#                       firmware so 0x1F00030000 is accessible (otherwise
#                       it data-aborts).
#   dtparam=uart0=on -> mux GPIO14/15 to RP1 UART0 and enable it.
enable_uart=1
uart_2ndstage=1
dtparam=uart0=on
pciex4_reset=0

# Legacy HDMI framebuffer for the on-screen colour-cycle + banner.  No
# KMS overlay (KMS would own the display and our mailbox framebuffer
# would never be scanned out).  depth=32 works around the Pi 5 firmware
# 16/32-bit framebuffer bug.
hdmi_force_hotplug=1
framebuffer_width=640
framebuffer_height=480
framebuffer_depth=32
framebuffer_ignore_alpha=1
max_framebuffers=1
disable_overscan=1
