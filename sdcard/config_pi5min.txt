# config.txt — Xinu Pi 5 first-light (minimal serial bring-up).
#
# To restore Raspberry Pi OS: copy config.txt.linux.bak back over
# config.txt (the original Linux config is backed up alongside).

arm_64bit=1
kernel=kernel_xinu.img
kernel_address=0x80000

# Route serial to the dedicated 3-pin DEBUG UART connector on the Pi 5
# (BCM2712 debug UART @ 0x107d001000 — the small JST-SH header next to
# the PCIe FFC).  This is the PCIe-independent UART the firmware itself
# logs to; enable_uart locks its reference clock so 115200 8N1 is stable.
enable_uart=1
uart_2ndstage=1

# No initramfs (that's a Linux leftover) — the first-light kernel is a
# flat ARM64 image.
auto_initramfs=0

# HDMI framebuffer for the on-screen banner mirror.  We do NOT load the
# KMS overlay (vc4-kms-v3d): KMS would own the display and our mailbox-
# allocated framebuffer would never be scanned out (the classic Pi 5
# "rainbow -> black on hand-off").  Instead we ask the firmware for a
# legacy simple framebuffer the kernel can paint directly.  depth=32 is
# the documented workaround for the Pi 5 firmware's 16/32-bit FB bug.
hdmi_force_hotplug=1
framebuffer_width=640
framebuffer_height=480
framebuffer_depth=32
framebuffer_ignore_alpha=1
max_framebuffers=1
disable_overscan=1
