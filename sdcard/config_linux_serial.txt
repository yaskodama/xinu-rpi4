# config.txt — stock Raspberry Pi OS (Pi 5) + serial console enabled.
# Used to VALIDATE the dedicated debug-UART wiring: if Linux boot logs
# appear on the 3-pin debug connector, the serial path is good and the
# problem is purely the Xinu kernel.  No kernel= line -> firmware loads
# the Linux kernel_2712.img.
dtparam=audio=on
camera_auto_detect=1
display_auto_detect=1
auto_initramfs=1
dtoverlay=vc4-kms-v3d
max_framebuffers=2
disable_fw_kms_setup=1
arm_64bit=1
disable_overscan=1
arm_boost=1

# Enable serial. enable_uart locks the UART clock; dtparam=uart0=on
# muxes GPIO14/15 (40-pin header pins 8/10) to RP1 UART0 so the Linux
# console (cmdline.txt: console=serial0,115200) reaches the cable the
# user wired there.
enable_uart=1
uart_2ndstage=1
dtparam=uart0=on

[cm4]
otg_mode=1

[cm5]
dtoverlay=dwc2,dr_mode=host

[all]
